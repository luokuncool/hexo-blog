```mysql
DROP FUNCTION IF EXISTS F_TEST
CREATE FUNCTION F_TEST(PID INT) RETURNS VARCHAR
BEGIN
  DECLARE NAME_FOUND VARCHAR DEFAULT "";
    SELECT EMPLOYEE_NAME INTO NAME_FOUND FROM TABLE_NAME WHERE ID = PID;
  RETURN NAME_FOUND;
END;
```
